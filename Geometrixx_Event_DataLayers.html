1. pageView

Trigger: Page load — fires on every page in the <head> before DOM render.

Found in: all HTML pages (index.html, login.html, 404error.html, product pages, checkout pages, etc.)

window.adobeDataLayer.push({
  event: "pageView",
  page: {
    pageName:      "Home Page",         // varies per page
    pageUrl:       window.location.href,
    pageType:      "content",            // "content" | "product" | "search" | "error"
    language:      "en",
    pageCategory:  "Home",               // varies per page
    SiteSection1:  "",                   // top-level section
    siteSection2:  "",                   // sub-section
    siteSection3:  "",                   // sub-sub-section
    referrer:      document.referrer,
    is404:         false                // true on 404error.html
  },
  user: GeoAuth.getUserDataLayer()   // see User Object below
});
PropertyTypeExample Values
page.pageNameString"Home Page", "Men:Fine Apparel:Blue Shirt", "Checkout:Shopping Cart", "404 Error", "Search Results"
page.pageTypeString"content", "product", "search", "error"
page.pageCategoryString"Home", "Men", "Women", "Checkout", "Company", "Search", "Error", "Login", "Signup"
page.SiteSection1String"Men", "Women", "Checkout", "Company", "Login", "Signup", ""
page.siteSection2String"Fine Apparel", "Activewear", "Shopping Cart", "Order Confirmation Step 4", ""
page.siteSection3String"Blue Shirt", "Tuxedo", ""
page.is404Booleanfalse (true only on 404error.html)

2. productView

Trigger: DOMContentLoaded on product detail pages (pageType === "product"). Extracts product info from DOM.

Found in: scripts/menu.js — fires on pages like blueshirt.html, tuxedo.html, etc.

window.adobeDataLayer.push({
  event: "productView",
  product: {
    id:       "GEO_BLUSHIRT",        // SKU from DOM
    name:     "Blue Shirt",           // product name from DOM
    price:    39.99,                  // parsed from DOM
    currency: "USD"
  }
});
PropertyTypeDescription
product.idStringProduct SKU extracted from page content
product.nameStringProduct name from .search_title element
product.priceNumberPrice parsed from "Price: $X.XX" pattern
product.currencyStringAlways "USD"

3. addtocart

Trigger: User clicks "Add to Cart" button on a product page.

Found in: scripts/simpleCart.js

window.adobeDataLayer.push({
  event: "addtocart",
  product: {
    id:       "GEO_BLUSHIRT",        // SKU from button value
    name:     "Blue Shirt",
    price:    39.99,
    currency: "USD",
    quantity: 1
  }
});
PropertyTypeDescription
product.idStringSKU from the add-to-cart button value
product.nameStringProduct name
product.priceNumberUnit price
product.currencyStringAlways "USD"
product.quantityNumberAlways 1 per click

4. cartOpen

Trigger: First "Add to Cart" action per session (fires once per session).

Found in: scripts/simpleCart.js

window.adobeDataLayer.push({
  event: "cartOpen",
  cart: {
    id:            "CART_1717600000_abc123",
    totalValue:    39.99,
    totalQuantity: 1,
    currency:      "USD"
  },
  product:       [/* array of cart products */]
});
PropertyTypeDescription
cart.idStringSession-generated cart ID (CART_{timestamp}_{random})
cart.totalValueNumberTotal cart value at time of event
cart.totalQuantityNumberTotal item count
cart.currencyStringAlways "USD"
product[]ArrayArray of product objects {name, id, price, quantity, category}

5. cartView

Trigger: DOMContentLoaded on the Shopping Cart page (when cart has items).

Found in: checkout/shoppingcart.html

window.adobeDataLayer.push({
  event: "cartView",
  cart: {
    id:            "CART_1717600000_abc123",
    currency:      "USD",
    totalPrice:    129.97,
    totalQuantity: 3
  },
  product: [
      { name: "Blue Shirt", id: "GEO_BLUSHIRT", price: 39.99, quantity: 1, category: "Apparel" }
  ]
});
PropertyTypeDescription
cart.idStringSession cart ID
cart.currencyStringAlways "USD"
cart.totalPriceNumberSum of all items × quantities
cart.totalQuantityNumberTotal item count in cart
product[].nameStringProduct name
product[].idStringProduct SKU
product[].priceNumberUnit price
product[].quantityNumberQuantity in cart
product[].categoryStringAlways "Apparel"

8. order confirmation

Trigger: Order confirmation page load (after successful purchase). Fires on both credit card and COD confirmation pages.

Found in: checkout/confirmreceipt4.html, checkout/confirmreceipt_cod.html

window.adobeDataLayer.push({
  event: "order confirmation",
  order: {
    id:             "ORD_1717600000_abc123",
    currency:       "USD",
    totalPrice:     129.97,
    totalQuantity:  3,
    shippingMethod: "Standard",           // or "COD"
    postalCode:     "90210",
    paymentMethod:  "Visa"               // or "COD"
  },
  product: [
      { name: "Blue Shirt", id: "GEO_BLUSHIRT", price: 39.99, quantity: 1, category: "Apparel" }
  ]
});
PropertyTypeDescription
order.idStringGenerated order ID from session storage
order.currencyStringAlways "USD"
order.totalPriceNumberOrder total
order.totalQuantityNumberTotal items ordered
order.shippingMethodStringDelivery type selected; defaults to "COD" on COD page
order.postalCodeStringShipping postal code
order.paymentMethodStringPayment method; hard-coded "COD" on COD page
product[]ArraySame product structure as cartView

9. search

Trigger: Search Results page load when a valid search term exists in the URL.

Found in: search/searchresults.html

window.adobeDataLayer.push({
  event: "search",
  search: {
    searchTerm:   "blue shirt",
    searchType:   "internal",       // determined from referrer
    method:       "enter"            // from URL param or default "enter"
  }
});
PropertyTypeDescription
search.searchTermStringUser's search query
search.searchTypeString"internal", "search results", "search engine", "women", "men", "checkout", "company", "login", "signup", "404 error", "home"
search.methodStringSearch method — "enter" (default) or from URL parameter

13. link click

Trigger: User clicks global CTAs (header links), footer CTAs, or menu navigation links.

Found in: scripts/menu.js

window.adobeDataLayer.push({
  event: "link click",
  link: {
    linkName:     "Shopping Cart",    // link text content
    linkLocation: "Home Page",        // current page name from pageView datalayer
    linkType:     "internal"          // "internal" | "external"
  }
});
PropertyTypeDescription
link.linkNameStringThe anchor's visible text content
link.linkLocationStringCurrent page's pageName (from pageView event)
link.linkTypeString"internal" (same hostname or # hrefs) or "external"

Shared: User Object — GeoAuth.getUserDataLayer()

Used by: pageView, userLogin, userLogout events.

Defined in: scripts/auth.js

// Logged-in user
{
  userId:        "USR_001",
  loginStatus:   "loggedin",
  customerType:  "consumer",
  loyaltyStatus: "guest",
  loyaltyTier:   "",
  country:       "US"
}

// Logged-out / anonymous user
{
  userId:        "",
  loginStatus:   "loggedOut",
  customerType:  "consumer",
  loyaltyStatus: "guest",
  loyaltyTier:   "",
  country:       "US"
}
PropertyTypeDescription
user.userIdStringUser ID from auth system; empty if not logged in
user.loginStatusString"loggedin" or "loggedOut"
user.customerTypeStringDefaults to "consumer"
user.loyaltyStatusStringDefaults to "guest"
user.loyaltyTierStringLoyalty tier; empty if none
user.countryStringDefaults to "US"
